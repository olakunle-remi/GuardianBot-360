    class MockGPIO:
        BCM = 1
        OUT = 1
        HIGH = 1
        LOW = 0

        def setup(self, pin, mode): print(f"Mock: Pin {pin} setup as output")
        def output(self, pin, value): print(f"Mock: Pin {pin} set to {value}")
    GPIO = MockGPIO()
else:
    import RPi.GPIO as GPIO

# --- Core Logic ---
def initialize_system():
    print("Initializing GuardianBot-360...")
    GPIO.setmode(GPIO.BCM)
    GPIO.setup(17, GPIO.OUT) # Assuming pin 17 is a motor/LED

def run_patrol_loop():
    print("Patrol Mode Active. Press Ctrl+C to stop.")
    try:
        while True:
